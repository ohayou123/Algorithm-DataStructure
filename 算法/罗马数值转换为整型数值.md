//将罗马数值转换为整型数值：
class Solution {
    public int romanToInt(String s) {
        if(s.length() == 0)
            return 0;
        int result = 0;
        int sub = getRomanValue(s.charAt(0));
        int lastv = sub;
        for(int i = 1;i < s.length();i ++)
        {
            int curv = getRomanValue(s.charAt(i));
            if(curv == lastv)
                sub += curv ;
            else if(curv < lastv)
            {
                result += sub ;
                sub = curv;
            }
                
            else
            {
                sub = curv - sub;
            }
                
            lastv = curv;
        }
        result += sub;
        return result;        
    }
    
    public static int getRomanValue(char c)
    {
        switch(c)
        {
            case 'I':return 1;
            case 'V':return 5;
            case 'X':return 10;
            case 'L':return 50;
            case 'C':return 100;
            case 'D':return 500;
            case 'M':return 1000;
            default:return 0;        
        }   
    }
}
